from collections import defaultdict

def readblocks(data):
    copy = False
    n = 0
    block = []
    chunk = defaultdict(lambda:[])
    for i in data:
        try:
            if "~~~~~~~~~~~~~~~~~~~~~~~~~~  #" in i:
                copy = True
            elif "文章编号:" in i:
                ids = i.replace('文章编号: ', '')
                source = block[0].split('|')[0]
                info = block[1]
                title = deletetab(block[3]) # 
                body = [j for j in block[6:] if j != '\n']
                body = ' '.join(body)
                body = deletetab(body)
                body = '"' + body  + '"'
                line = '\t'.join([ids, source, info, title, body])
                chunk[ids] = line
                block = []
                n += 1
                if n%10 == 0:
                    flushPrint(n)
                copy = False
            # copy must be here.
            elif copy:
                block.append(i)
        except Exception as e:
            print(i, e)
            pass
    return chunk
